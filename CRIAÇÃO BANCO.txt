CREATE TABLE CENTRODECUSTOPAI(
	ID_CENTRODECUSTOPAI SERIAL NOT NULL PRIMARY KEY,
	DESCRICAO VARCHAR(45),
	CODIGOPAI VARCHAR(2)
);

CREATE TABLE CENTRODECUSTOFILHO(
	ID_CENTRODECUSTOFILHO SERIAL NOT NULL PRIMARY KEY,
	DESCRICAO VARCHAR(45),
	CODIGOFILHO VARCHAR(6)
);

create table RESUMOLANCAMENTOS(
id_lancamento SERIAL not null primary key,	
valor double precision,
ID_CENTRODECUSTOPAI  integer,
ID_CENTRODECUSTOFILHO integer,	
	
FOREIGN KEY (ID_CENTRODECUSTOPAI ) REFERENCES CENTRODECUSTOPAI (ID_CENTRODECUSTOPAI),
FOREIGN KEY (ID_CENTRODECUSTOFILHO) REFERENCES CENTRODECUSTOFILHO (ID_CENTRODECUSTOFILHO)
);
